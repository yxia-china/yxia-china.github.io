    #!/bin/bash

    echo ${MYID:-1} > /zookeeper/data/myid

    if [ -n “$SERVERS” ]; then
    IFS=, read -a servers <<<”$SERVERS”
    for i in “${!servers[@]}”; do
    printf “\nserver.%i=%s:2888:3888” “$((1 + $i))” “${servers[$i]}” >> /zookeeper/conf/zoo.cfg
    #输出 server.1~3=zookeeper1-3:2888:3888
    done
    fi

    cd /zookeeper
    exec “$@”
